import React, { useState } from "react";
import IncomeForm from "./IncomeForm";
import ExpenseForm from "./ExpenseForm";
import Summary from "./Summary";

const BudgetTracker = () => {
  const [income, setIncome] = useState(0);
  const [expenses, setExpenses] = useState(0);

  const addIncome = (amount: number) => {
    if (amount > 0) {
      setIncome((prev) => prev + amount);
    }
  };

  const addExpense = (amount: number) => {
    if (amount > 0) {
      setExpenses((prev) => prev + amount);
    }
  };

  const resetAll = () => {
    setIncome(0);
    setExpenses(0);
  };

  return (
    <div>
      <h2>ðŸ’° Budget Balance Assistant</h2>
      <IncomeForm onAddIncome={addIncome} />
      <ExpenseForm onAddExpense={addExpense} />
      <Summary income={income} expenses={expenses} onReset={resetAll} />
    </div>
  );
};

export default BudgetTracker;

import React, { useState } from "react";

const IncomeForm = ({ onAddIncome }) => {
  const [amount, setAmount] = useState("");

  const handleSubmit = (e) => {
    e.preventDefault();
    const value = Number(amount);
    if (value > 0) {
      onAddIncome(value);
      setAmount("");
    }
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        data-testid="income-input"
        type="number"
        value={amount}
        onChange={(e) => setAmount(e.target.value)}
        placeholder="Enter income"
      />
      <button data-testid="add-income-btn" type="submit">
        Add Income
      </button>
    </form>
  );
};

export default IncomeForm;

import React from "react";

const Summary = ({ income, expenses, onReset }) => {
  const balance = income - expenses;

  return (
    <div>
      <p data-testid="total-income">Total Income: {income}</p>
      <p data-testid="total-expenses">Total Expenses: {expenses}</p>
      <p data-testid="balance">Balance: {balance}</p>
      <button data-testid="reset-btn" onClick={onReset}>
        Reset
      </button>
    </div>
  );
};

export default Summary;
